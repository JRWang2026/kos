    var m = Math.floor(sec / 60);
    var s = sec % 60;
    return m + ':' + (s < 10 ? '0' + s : s);
  }

  // Countdown timer
  setInterval(function() {
    remaining--;
    if (remaining < 0) {
      remaining = REFRESH_SECONDS;
      location.reload();
      return;
    }
    if (countdownEl) countdownEl.textContent = formatTime(remaining);
  }, 1000);

  // Manual refresh
  if (refreshBtn) {
    refreshBtn.addEventListener('click', function() {
      refreshBtn.textContent = '⟳ 刷新中…';
      location.reload();
    });
  }

  // Toggle all jobs — lazy load from JSON on first expand
  if (toggleBtn && allJobsEl) {
    toggleBtn.addEventListener('click', function() {
      expanded = !expanded;
      allJobsEl.style.display = expanded ? 'block' : 'none';
      toggleBtn.textContent = expanded ? '收起' : '显示全部岗位';

      if (expanded && !allJobsLoaded) {
        allJobsLoaded = true;
        fetch('/data/seek-nz/latest.json?t=' + Date.now(), { cache: 'no-store' })

            }
            if (allJobsBody) allJobsBody.innerHTML = html;
          })
          .catch(function() {
            if (allJobsBody) allJobsBody.innerHTML = '<tr><td colspan="3">Failed to load</td></tr>';
          });
      }
    });
  }
